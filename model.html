📝 การสร้าง Model Class

เราสร้าง Neural Network โดย สืบทอด จาก nn.Module และกำหนด layers ใน __init__ ส่วน forward กำหนดว่าข้อมูลไหลอย่างไร

class NeuralNetwork(nn.Module): # 1. กำหนด layers def __init__(self): super().__init__() self.flatten = nn.Flatten() self.linear_relu_stack = nn.Sequential( nn.Linear(28*28, 512), # Hidden 1 nn.ReLU(), nn.Linear(512, 512), # Hidden 2 nn.ReLU(), nn.Linear(512, 10), # Output ) # 2. กำหนดการไหลของข้อมูล def forward(self, x): x = self.flatten(x) logits = self.linear_relu_stack(x) return logits # สร้าง model model = NeuralNetwork()
