📥 การโหลด Dataset

# Import สิ่งที่ต้องการ
import torch
from torchvision import datasets
from torchvision.transforms import ToTensor

# โหลด Training Data
training_data = datasets.FashionMNIST(
    root="data",      root
    train=True,         train
    download=True,       download
    transform=ToTensor() transform
)

# โหลด Test Data
test_data = datasets.FashionMNIST(
    root="data",
    train=False,
    download=True,
    transform=ToTensor()
)

📌 พารามิเตอร์สำคัญ:

root ที่เก็บข้อมูล
train True=เทรน, False=ทดสอบ
download ดาวน์โหลดถ้ายังไม่มี
transform แปลงข้อมูล (เช่น เป็น Tensor)

🔍 การเข้าถึงข้อมูล

เข้าถึงเหมือน list: training_data[index]

# ดึงข้อมูลตัวที่ 0
img, label = training_data[0]

print(img.shape)   # torch.Size([1, 28, 28])
print(label)        # 9
print(len(training_data)) # 60000

🏗️ สร้าง Custom Dataset

Dataset ที่สร้างเองต้องมี 3 ฟังก์ชัน:

1__init__

รันครั้งเดียวตอนสร้าง object

def __init__(self, annotations_file, img_dir, transform=None, target_transform=None): self.img_labels = pd.read_csv(annotations_file) # อ่าน labels self.img_dir = img_dir # โฟลเดอร์รูป self.transform = transform # แปลงรูป

2__len__

คืนค่าจำนวนข้อมูลทั้งหมด

def __len__(self): return len(self.img_labels)

3__getitem__

โหลดและคืนข้อมูลที่ index

def __getitem__(self, idx): img_path = os.path.join(self.img_dir, self.img_labels.iloc[idx, 0]) image = decode_image(img_path) # อ่านรูป label = self.img_labels.iloc[idx, 1] # อ่าน label if self.transform: image = self.transform(image) return image, label

🔄 DataLoader คืออะไร?

DataLoader ช่วยจัดการ:

  • 📦 แบ่ง Batch - จัดกลุ่มข้อมูลเป็น batch
  • 🔀 Shuffle - สุ่มข้อมูลทุก epoch
  • ⚡ Parallel - โหลดข้อมูลพร้อมกัน

Code:

from torch.utils.data import DataLoader

train_dataloader = DataLoader(
    training_data,      dataset
    batch_size=64,   batch_size
    shuffle=True,     shuffle
    num_workers=2      num_workers
)
